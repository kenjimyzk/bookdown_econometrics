12.1 Pooled OLS

simdata0 <- function(N = 100) {
  x <- rnorm(N, 10, 10)
  y <- 1 + x + rnorm(N, 1, 10)
  data.frame(id = 1:N, x, y)
}

df0 <- data.frame(simdata0(), time = 0)
df1 <- data.frame(simdata0(), time = 1)
df2 <- data.frame(simdata0(), time = 2)
df <- bind_rows(df0, df1, df2)
head(df)
df <- pdata.frame(df, index = c("id", "time"))
fm.pooling <- plm(y ~ x, data = df, model = "pooling")
